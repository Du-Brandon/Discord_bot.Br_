```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int maxn = 2e5 + 5;
string s;
vector<int> vec[26];
ll pre[maxn], ans = 0;

ll calc(int idx) {
    int n = vec[idx].size();
    memset(pre, 0, sizeof(pre));
    ll ret = 0;
    for (int i = 1; i < n; i++) {
        pre[i - 1] = vec[idx][i] - vec[idx][i - 1] - 1;
    }
    ret += pre[0];
    for (int i = 1; i < n - 1; i++) {
        pre[i] += pre[i - 1] + 1;
        ret += pre[i];
    }
    return ret;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    cin >> s;
    for (int i = 0; i < s.length(); i++)
        vec[s[i] - 'a'].push_back(i);

    for (int i = 0; i < 26; i++) {
        ans += calc(i);
    }

    cout << ans << '\n';

    return 0;
}
```
